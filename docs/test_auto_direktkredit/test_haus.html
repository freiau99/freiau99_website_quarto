library(htmltools)

# =====================================================
# ZAHLEN ANPASSEN
# =====================================================

dk_gesamt      <- 280000
dk_abbezahlt   <- 150000

bank_gesamt    <- 500000
bank_abbezahlt <- 120000

# =====================================================
# BERECHNUNGEN
# =====================================================

gesamt <- dk_gesamt + bank_gesamt
abbezahlt <- dk_abbezahlt + bank_abbezahlt

dk_quote <- min(dk_abbezahlt / dk_gesamt, 1)
bank_quote <- min(bank_abbezahlt / bank_gesamt, 1)

# Dachziegel
ziegel_total <- 20
ziegel_gruen <- round(ziegel_total * dk_quote)

# Fenster
fenster_total <- 40
fenster_blau <- round(fenster_total * bank_quote)

# =====================================================
# DACHZIEGEL
# =====================================================

ziegel_nr <- 1

rows <- c(7, 6, 4, 3)

for(r in seq_along(rows)) {

  n_tiles <- rows[r]

  y <- 120 - (r - 1) * 18

  total_width <- n_tiles * 28

  start_x <- 250 - total_width / 2

  for(col in 1:n_tiles) {

    klasse <- if (ziegel_nr <= ziegel_gruen) {
      "tile-paid"
    } else {
      "tile-open"
    }

    x <- start_x + (col - 1) * 28

    svg <- c(
      svg,
      glue(
        '
        <rect
          x="{x}"
          y="{y}"
          width="24"
          height="12"
          rx="2"
          class="{klasse} tile"
          clip-path="url(#roofClip)"
        />
        '
      )
    )

    ziegel_nr <- ziegel_nr + 1
  }
}
